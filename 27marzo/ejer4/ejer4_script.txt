!-------------
!Switch MLS
!-------------
en
config t

hostname MLS

enable password class

line con 0
password cisco
login
logging synchronous

line vty 0 15
password cisco
login
logging synchronous

! Habilitar ruteo en switch multicapa
ip routing

! CONFIGURAR EL PUERTO G0/2
int g0/2
! indicar que no es un puerto de switch, sino un puerto de router
no switchport
ip address 209.165.200.225 255.255.255.252
no shut

! CONFIGURAR VLANs
vlan 10
name Staff
exit

vlan 20
name Students
exit

vlan 30
name Faculty
exit

! CONFIGURAR SUBINTERFACES = CONFIGURAR PUERTOS TRONCALES
! MLS = Simplemente indicas que el puerto de subinterfaces sea un puerto troncal
! MLS = Usa el puerto troncal y necesita las interfaces LOGICAS

! INTEFAZ TRONCAL PARA SWITCH MULTICAPA (MLS)
int g0/1
switchport trunk encapsulation dot1q
switchport mode trunk
no shut

! CONFIGURAR IP DEL SWITCH -> ! AÑADIR INTERFACES LOGICAS
int vlan 1
ip address 192.168.99.254 255.255.255.0
! SIEMPRE NO SHUT EN VLAN 1
no shut

int vlan 10
ip address 192.168.10.254 255.255.255.0

int vlan 20
ip address 192.168.20.254 255.255.255.0

int vlan 30
ip address 192.168.30.254 255.255.255.0

! CONFIGURAR RUTA POR DEFAULT HACIA INTERNET
ip route 0.0.0.0 0.0.0.0 g0/2
