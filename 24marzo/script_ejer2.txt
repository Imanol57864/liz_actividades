config t
hostname RFrontera

no ip domain-lookup
service password-encryption 

! guarda el password "class" para la terminal de configuración (encriptada)
enable secret class
! guarda el password "class" para la terminal de configuración (sin encriptar)
! enable password class

banner motd # Unauthorized access is strictly prohibited. # 

! configuracion de la terminal, permite configurar de manera más flexible "> ... ", con contrasena cisco
line con 0
password cisco
login
logging synchronous

! numero de conexiones al router (5 conexiones), con contrasena cisco
line vty 0 4 
password cisco 
login
logging synchronous
exit

int s0/0/1
desc Interfaz que conecta con ISP
ip address 65.255.255.253 255.255.255.252
no shut

! Declaracin de subinterfaces
! bite critico -2 porque la de brodcast NO se usa para interfaces
! Siempre se requiere de una nativa, la cual será entre el router y los switches primarios
! Son interfaces logicas, se pegan a una intefaz fisica, la fisica si se levanta
! Se coloco la ip_ultima_valida para las interfaces logicas porque asi indica el ejer
! Las interfaces logicas requieren de una ip valida, ya que sirven como una ip_bloque
int g0/0.1
desc vlan 1 nativa
encapsulation  dot1q  1
ip add 192.168.1.158 255.255.255.248

int g0/0.10
desc vlan 10 manager
encapsulation  dot1q  10
ip add 192.168.1.142 255.255.255.240


int g0/0.20
desc vlan 20 users
encapsulation  dot1q  20
ip add 192.168.1.126 255.255.255.128


int g0/0.30
desc vlan 30 services
encapsulation  dot1q  30
ip add 192.168.1.150 255.255.255.248

! no requiere de ip
int g0/0
no shut

! Establecer una ruta por default que se encargue de sacar el tr fico a Internet
ip route 0.0.0.0 0.0.0.0 s0/0/1

! next hop
! ip route 0.0.0.0 0.0.0.0 65.255.255.252

end

no shut

end
