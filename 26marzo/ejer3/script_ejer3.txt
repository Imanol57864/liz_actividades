hostname R1

service password-encryption
no ip domain-lookup

enable password class

line con 0
password cisco
login
logging synchronous

line vty 0 4
password cisco
login
logging synchronous

banner motd # Solo acceso autorizado #

interface G0/0
description Enlace a LAN Amarilla
ip address 192.168.1.1 255.255.255.0
no shut

! Enlace DCE
interface S0/0/0
description Enlace a R2
ip address 192.168.12.1 255.255.255.252
clock rate 128000
no shut

interface S0/0/1
description Enlace a R3
ip address 192.168.13.1 255.255.255.252
no shut

! OSPF (Para todas las redes conectadas)
router ospf 8
! ip de subred || mascara comodin || area 0
network 192.168.1.0 0.0.0.255 area 0
network 192.168.12.0 0.0.0.3 area 0
network 192.168.13.0 0.0.0.3 area 0
! passive-interface para denegar la 4rd-Saturacion a donde no
passive-interface g0/0

! A través de OSPF distribuir la ruta DEFAULT DIRECTAMENTE a los demás routers

! Ruta DEFAULT DIRECTAMENTE para el RFRONTERA -> INTERNET
! 0.0.0.0 || 0.0.0.0 || INTERFAZ_SALIDA

! 0.0.0.0 || 0.0.0.0 || IP_INTERFAZ_LLEGADA
! ip route 0.0.0.0 0.0.0.0 200.6.10.193
