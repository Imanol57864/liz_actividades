en
config t

hostname R1

service password-encryption
no ip domain-lookup

enable password class

line con 0
password cisco
login
logging synchronous

line vty 0 4
password cisco
login
logging synchronous

banner motd # Solo acceso autorizado #

interface G0/0
description Enlace a Red Amarilla
ip address 192.168.1.1 255.255.255.0
! SE COLOCA EN LA INTERFAZ DE CARA A LA SUBNET
! INTERFAZ PEDIRÁ DHCP AL SERVIDOR DHCP (ROUTER RF)
! ip helper-address IP_INTERFAZ_SERVIDORDHCP
ip helper-address 192.168.12.2 
no shut

interface S0/0/0
description Enlace a RF
ip address 192.168.12.1 255.255.255.252
clock rate 128000
no shut

router ospf 1
network 192.168.1.0 0.0.0.255 area 0
network 192.168.12.0 0.0.0.3 area 0
passive-interface g0/0

end
