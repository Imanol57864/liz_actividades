en
config t

! TUNNEL != INTERFAZ != SUBNET
! TUNNEL = LA IP DE LA CONEXION LOGICA
! INTERFAZ = LA INTERFAZ DE SALIDA DEL ROUTER AL ISP/INTERNET
! SUBNET = LA SUBRED AL QUE EL ROUTER OTORGA INTERNET

! interfaz para el tunnel
interface tunnel 0
! ip address IP_TUNEL_SALIDA MASK_TUNEL_SALIDA
ip address 192.168.1.1 255.255.255.252
! tunnel source INTERFAZ_TUNNEL_SALIDA
tunnel source G0/0/0
! tunne destination IP_INTERFAZ_LLEGADA
tunnel destination 200.0.0.2
tunnel mode gre ip
no shut

! ruta por defecto al isp
ip route 0.0.0.0 0.0.0.0 100.0.0.1

! protoco de ruteo ! para r1 conozca OFFICE B
router ospf 1
network 192.168.1.0 0.0.0.3 area 0
network 10.0.1.0 0.0.0.255 area 0
passive-interface g0/0
passive g0/0/0

! XD 
! RUTA DE TRAFICO PRIVADO
! ip route IP_SUBNET_LLEGADA MASK_SUBNET_LLEGADA IP_TUNNEL_LLEGADA
!ip route 10.0.2.0 255.255.255.0 192.168.1.2
